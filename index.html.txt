    document.getElementById('submitPost').addEventListener('click', function() {
      const videoLink = document.getElementById('videoLink').value;
      const desc = document.getElementById('description').value;

      if (!videoLink || !desc) {
        alert("Lütfen video linki ve açıklama girin.");
        return;
      }

      // YouTube embed link yap
      let embedLink = videoLink;
      if (videoLink.includes("youtube.com/watch")) {
        const url = new URL(videoLink);
        const v = url.searchParams.get("v");
        embedLink = `https://www.youtube.com/embed/${v}?autoplay=0`;
      } else if (videoLink.includes("youtu.be/")) {
        const id = videoLink.split('/').pop();
        embedLink = `https://www.youtube.com/embed/${id}?autoplay=0`;
      }

      // Yeni post oluştur
      const post = document.createElement('div');
      post.className = 'post';
      post.innerHTML = `
        <div class="video-placeholder">
